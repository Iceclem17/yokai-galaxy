(function() {
    // --- CONFIGURATION ---
    const nomDeBase = "qr-piece-rouge"; // Change ce nom ici
    const padding = 16;               // Taille de la bordure blanche
    // ---------------------

    const items = document.querySelectorAll('.qr-item');
    
    if (items.length === 0) {
        console.error("Aucun QR code trouvé.");
        return;
    }

    console.log(`Déchargement de ${items.length} QR codes avec le nom : ${nomDeBase}...`);

    items.forEach((item, index) => {
        const img = item.querySelector('img');

        if (img) {
            // On crée le nom : "nom-de-base-1", "nom-de-base-2", etc.
            const numero = index + 1;
            const fileName = `${nomDeBase}-${numero}.png`;

            // Création du canvas pour la bordure blanche
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');

            canvas.width = img.naturalWidth + (padding * 2);
            canvas.height = img.naturalHeight + (padding * 2);

            // Fond blanc
            ctx.fillStyle = "#ffffff";
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Dessin du QR code centré
            ctx.drawImage(img, padding, padding);

            // Création du lien et téléchargement
            const link = document.createElement('a');
            link.href = canvas.toDataURL("image/png");
            link.download = fileName;

            // Délai pour éviter de bloquer le navigateur
            setTimeout(() => {
                link.click();
                console.log(`[${numero}/${items.length}] Téléchargé : ${fileName}`);
                
                if (numero === items.length) {
                    alert(`Terminé ! Tes ${items.length} images sont prêtes.`);
                }
            }, index * 300);
        }
    });
})();
